  6def prop_subquery(
  7        record: dict, 
  8        suffix : str = "", 
  9        exclude_keys: list[str] = []
 10        )-> (str, dict):
 11    """
 12    Generates a Cypher substring statement to set properties for a record, excluding given keys
 13
 14    Args:
 15        record: Dict of Node or Relatioship properties to convert
 16
 17        suffix: String suffix to make parameter values unique
 18
 19        exclude_keys: List of dictionary key values to ignore from substring generation
 20
 21    
 22    Returns:
 23        A tuple containing the substring and a dict of parameters
 24    """
 25
 26    # Validate
 27    if isinstance(record, dict) == False:
 28        return ("", {})
 29    if len(record) == 0:
 30        return ("", {})
 31    
 32    # Embed any prop data within brackets { }
 33    params = {}
 34    query = " {"
 35
 36    filtered_keys = [key for key in record.keys() if key not in exclude_keys]
 37    sorted_keys = sorted(list(filtered_keys))
 38
 39    for idx, a_key in enumerate(sorted_keys):
 40        value = record[a_key]
 41
 42        # Do not set properties with a None/Null/Empty value
 43        if value is None:
 44            continue
 45        if isinstance(value, str):
 46            if value.lower() == "none":
 47                continue
 48            if value.lower() == "null":
 49                continue
 50            if value.lower() == "empty":
 51                continue
 52            if value.lower() == "":
 53                continue
 54
 55        # Prefix multiple items in Cypgher with comma
 56        if idx!= 0:
 57            query += ", "
 58
 59        # Mark value for parameterization and add to params return dict
 60        param_key = f'{a_key}_{suffix}'
 61        params[param_key] = value
 62
 63        # Cypher string query requires { } to designate parameter values
 64        query += f'`{a_key}`:${param_key}'
 65
 66    # Close out query
 67    query += "}"
 68
 69    return query, params

 84    """
 85    Returns elements to be added into a batch node creation query.

 98    """

126def upload_node_records_query(
127    label: str,
128    nodes: list[dict],
129    node_key: str = "_uid",
130    dedupe : bool = True
131    )->(str, dict):
132    """
133    Generate Cypher Node update query.
134
135    Args:
136        label: Label of Nodes to generate
137
138        nodes: A list of dictionaries representing Node properties
139
140        dedupe: Remove duplicate entries. Default True.
141    
142    Returns:
143        A tuple with the full Cypher query statment and dictionary of parameters to pass to the Neo4j driver
144
145    Raises:
146        Exceptions if data is not in the correct format or if the upload fails.
147    """
148        
149    if nodes is None:
150        return ("", {})
151    if len(nodes) == 0:
152        return ("", {})
153    
154    logger.debug(f'Starting to process node labeled: {label} ...')
155
156    query = ""
157
158    with_elements, params = with_node_elements(
159        nodes,
160        node_key,
161        dedupe
162    )
163
164    if len(with_elements) is None:
165        logger.warning(f'Could not process nodes labeled {label}. Check if data exsists and matches expected schema')
166        return ("", {})
167
168    with_elements_str = ",".join(with_elements)
169
170    # Assemble final query
171    query = f"""WITH [{with_elements_str}] AS node_data\nUNWIND node_data AS node"""
172    
173    if dedupe == True:
174        query += f"\nMERGE (n:`{label}` {{`{node_key}`:node.`{node_key}`}})"
175    else:
176        query += f"\nCREATE (n:`{label}`)"
177    query += "\nSET n += node"
178
179    return query, params

182def upload_nodes(
183    neo4j_creds:(str, str, str),
184    nodes: dict,
185    node_key: str = "_uid",
186    database : str = "neo4j",
187    dedupe : bool = True,
188    max_batch_size: int = 500
189)-> (int, int):
190    """
191    Uploads a list of dictionary objects as nodes.
192
193    Args:
194        neo4j_credits: Tuple containing the hostname, username, and password of the target Neo4j instance
195
196        nodes: A dictionary of objects to upload. Each key is a unique node label and contains a list of records as dictionary objects.
197
198        node_key: String property key that uniquely identifies each node. Default '_uid'.
199
200        database: Name of Neo4j database to connect to. Default 'neo4j'.
201
202        dedupe: Remove duplicate entries. Default True
203
204        max_batched_size: Maximum number of nodes to upload in a single batch. Default 500.
205    
206    Returns:
207        A tuple containing the number of nodes created and properties set.
208
209    Raises:
210        Exceptions if data is not in the correct format or if the upload fails.
211    """
212    if nodes is None:
213        return 0, 0
214    if len(nodes) == 0:
215        return 0, 0
216    
217    # For reporting
218    nodes_created = 0
219    props_set = 0
220
221    query = """"""
222
223    logger.debug(f'Uploading node records: {nodes}')
224
225    # Sort Nodes to guarantee consistent output
226    filtered_keys = [key for key in nodes.keys()]
227    sorted_keys = sorted(list(filtered_keys))
228
229    # NOTE: Process each Node label separately so as to not timeout connection with db. Somewhere between 100-1000 Nodes.
230    for node_label in sorted_keys:
231
232        total_nodes_list = nodes.get(node_label, None)
233
234        # TODO: Make a calculation of node * properties for a better batch calculation
235
236        # Break relationships into batches of 500
237        b = max_batch_size
238        chunked_nodes_list = [total_nodes_list[i * b:(i + 1) * b] for i in range((len(total_nodes_list) + b - 1) // b )]  
239
240        for nodes_list in chunked_nodes_list:
241            if nodes_list is None:
242                logger.error(f'No values for node label {node_label} found. Skipping.')
243                continue
244
245            # Process all similar labeled nodes together
246            nodes_query, nodes_params = upload_node_records_query(
247                node_label, 
248                nodes_list, 
249                dedupe=dedupe, 
250                node_key=node_key)
251
252            logger.debug(f'upload nodes query: {query}')
253
254            records, summary, keys = execute_query(
255                neo4j_creds, 
256                nodes_query, 
257                params=nodes_params, 
258                database=database)
259
260            # Sample summary
261            # {'metadata': {'query': '<query>', 'parameters': {}, 'query_type': 'w', 'plan': None, 'profile': None, 'notifications': None, 'counters': {'_contains_updates': True, 'labels_added': 17, 'nodes_created': 17, 'properties_set': 78}, 'result_available_after': 73, 'result_consumed_after': 0}
262            try:
263                created = summary.counters.nodes_created
264                nodes_created += created
265            except Exception as e:
266                logger.debug(f'No nodes labeled: {node_label} created')
267
268            try:
269                props = summary.counters.properties_set
270                props_set += props
271            except Exception as e:
272                logger.debug(f'No node properties for nodes labeled: {node_label} created')
273            
274            # logger.info(f'Results from upload nodes: \n\tRecords: {records}\n\tSummary: {summary.__dict__}\n\tKeys: {keys}')
275    
276    return nodes_created, props_set

278def target_key(
279        record: dict,
280        prefix: str,
281        override_suffix: str) -> str:
282    
283    # Original implementation - now takes presidence with multiple options
284    explicit_key = f"{prefix}{override_suffix}"
285    explicit_value = record.get(explicit_key, None)
286    if explicit_value is not None:
287        return explicit_key
288    
289    # find all matching keys
290    keys = record.keys()
291    filtered_keys = [k for k in keys if k.startswith(prefix)]
292
293    if len(filtered_keys) == 0:
294        # No from key found!
295        return None
296    
297    # Take first if multiple matches
298    key = sorted(filtered_keys)[0]
299    return key

312def with_relationship_elements(
313    type: str,
314    relationships: list[dict],
315    nodes_key: str = "_uid",
316    dedupe: bool = True
317    ) -> (str, dict):
318    """
319    Returns elements to be added into a batch relationship creation query.
320
321    Args:
322        type: The relationship type.
323
324        relationships: A list of dictionary records for each relationship property.
325
326        nodes_key: The property key that uniquely identifies Nodes. 
327
328        dedupe: Should duplicate relationships be purged from final results. Default True.
329    
330    Returns:
331        A tuple of the query substring and parameters dictionary
332    """
333    # TODO: Possible cypher injection entry point?
334
335    results_list = []
336    results_params = {}
337
338    if dedupe == True:
339        relationships = [dict(t) for t in {tuple(r.items()) for r in relationships}]
340
341    # Force sort bc list is not being consistent
342    # Little complicated looking because the from and to node id keys are now dynamic, and not necessarily the same for all relationships
343    relationships = sorted(relationships, key=lambda x: (x[from_key(x, nodes_key)], x[to_key(x, nodes_key)]))
344    
345    for idx, rel in enumerate(relationships):
346
347        suffix = f"r{idx}"
348
349        # Get unique key-value of from and to nodes
350        from_node_key = from_key(rel, nodes_key)
351        to_node_key = to_key(rel, nodes_key)
352        from_node_value = rel[from_node_key]
353        to_node_value = rel[to_node_key]
354
355        if from_node_value is None:
356            logger.warning(f'{type} Relationship from node missing target value for key: {from_key}')
357            continue
358        if to_node_value is None:
359            logger.warning(f'{type} Relationship to node missing target value for key: {to_key}')
360            continue
361
362
363        # Validate we from and to nodes to work with
364        if from_node_value is None:
365            logger.warning(f'{type} Relationship missing {from_key} property. Skipping relationship {rel}')
366            continue
367        if to_node_value is None:
368            logger.warning(f'{type} Relationship missing {to_key} property. Skipping relationship {rel}')
369            continue
370
371        # string and params for props
372        props_string, params = prop_subquery(
373            rel,
374            suffix=suffix,
375            exclude_keys=[from_node_key, to_node_key])
376
377        # Add from and to node identifiers as params
378        from_node_key_param = f"{from_node_key}_{suffix}"
379        to_node_key_param = f"{to_node_key}_{suffix}"
380
381        params[from_node_key_param] = from_node_value
382        params[to_node_key_param] = to_node_value
383
384        with_element = f"[${from_node_key_param},${to_node_key_param},{props_string}]"
385        results_list.append(with_element)
386        results_params = results_params | params
387    
388    return results_list, results_params

392def upload_relationship_records_query(
393        type: str,
394        relationships: list[dict],
395        nodes_key: str,
396        dedupe: bool = True,
397) -> (str, dict):
398
399        if relationships is None:
400            return ("", {})
401        if len(relationships) == 0:
402            return ("", {})
403        
404        logger.debug(f'Starting to process relationships type: {type} ...')
405
406        # Process all similar labeled nodes together
407        with_elements, params = with_relationship_elements(
408            type,
409            relationships, 
410            nodes_key=nodes_key, 
411            dedupe=dedupe)
412
413        if len(with_elements) is None:
414            logger.warning(f'Could not process relationships type {type}. Check if data exsists and matches expected schema')
415            return ("", {})
416        
417        with_elements_str = ",".join(with_elements)
418
419        # Assemble final query
420        rel_upload_query = f"""WITH [{with_elements_str}] AS from_to_data\nUNWIND from_to_data AS tuple\nMATCH (fromNode {{`{nodes_key}`:tuple[0]}})\nMATCH (toNode {{`{nodes_key}`:tuple[1]}})"""
421
422        # Merge only updates, creates new if not already existent. Create ALWAYS creates a new relationship
423        if dedupe == True:
424            rel_upload_query += f"\nMERGE (fromNode)-[r:`{type}`]->(toNode)"
425        else:
426            rel_upload_query += f"\nCREATE (fromNode)-[r:`{type}`]->(toNode)"
427        
428        # Update Relationship properties if any
429        rel_upload_query +=f"\nSET r += tuple[2]"
430
431        return rel_upload_query, params
